SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION,2);
SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION,1);

# --- New Code Block --- 

#version 330

layout (location = 0) in vec2 AttribVertex;
layout (location = 1) in vec2 AttribCoordUV;

out vec2 CoordUV;

void main(){
  mat4 MVP = mat4(1.0);

  gl_Position = MVP * vec4(AttribVertex,0.0,1000.0);
  CoordUV = AttribCoordUV;
}

# --- New Code Block --- 

#version 330

uniform sampler2D UniformTextura;

in  vec2 CoordUV;
out vec4 FragColor;

void main(){
  FragColor = texture(UniformTextura,CoordUV);
}
