#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(void){
    
    int rowIndex, colIndex, rowlength = 0, collength = 0;           /* rowlength -> no. of rows & collength -> no. of columns */
    printf("Enter number of rows and columns for matrix1\n");
    scanf("%d %d", &rowlength, &collength);
        
    double **matrix1 = malloc(rowlength * collength * sizeof(double));
    if(!matrix1){
        fprintf(stderr, "Memory allocation failed !!\n");
        return 1;
    }
    
    double  total_time;                         /* Variables for measuring total execution time */
    clock_t start_time, end_time;

    for(rowIndex = 0; rowIndex < rowlength; rowIndex++){
        for(colIndex = 0; colIndex < collength; colIndex++){
            matrix1[rowIndex][colIndex] = 0;            /* Initializing the matrix to 0 */
        }
    }
    
    start_time = clock();   
    printf("Matrix 1:\n");
    srand(1);
    for(rowIndex = 0; rowIndex < rowlength; rowIndex++){
        for(colIndex = 0; colIndex < collength; colIndex++){
            matrix1[rowIndex][colIndex] = (rand() + 0.11) * 0.000234; /* Enetring elements as random variables */

        }
    }
    for(rowIndex = 0; rowIndex < rowlength; rowIndex++){
        for(colIndex = 0; colIndex < collength; colIndex++){
            printf("%.2f\n", matrix1[rowIndex][colIndex]);

        }
    }
    
    
    free(matrix1);
    
    end_time = clock();
    total_time = ((double)(end_time - start_time))/ CLOCKS_PER_SEC;
    printf("Total execution time: %f seconds\n", total_time);

    return 0;
}
