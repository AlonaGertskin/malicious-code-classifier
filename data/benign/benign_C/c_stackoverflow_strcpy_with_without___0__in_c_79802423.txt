int main()
{
    const char *s = "this is string";

    char *hello = malloc(strlen(s) + 1);
    strcpy(hello, s);

    char *world = malloc(strlen(s));
    strcpy(world, s);

    printf("Hello: %s\n", hello);
    printf("World: %s\n", world);
    
    return 0;
}
