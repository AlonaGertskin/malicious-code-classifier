#include <arpa/inet.h>
#include <stdio.h>
#include <string.h>

int main() {
  int    bytes_sent;
  char   send_msg[100];
  struct sockaddr_in to_addr;
  memset(&to_addr, 0, sizeof(struct sockaddr_in));

  to_addr.sin_family      = AF_INET;
  to_addr.sin_port        = 5001;
  bytes_sent = sendto(socket(AF_INET,SOCK_DGRAM,0), send_msg, sizeof(send_msg), 0,
                      (struct sockaddr *)&to_addr, sizeof(to_addr));
  printf("res = %d bytes\n", bytes_sent);
  return 0;
}
