#include <stdio.h>

int main() {

    FILE *pRead = fopen("input.txt", "r");
    char buffer[1024] = {0};

    if (pRead == NULL) {
        printf("Could not open file\n");
        return 1;
    }

    while (fgets(buffer, sizeof(buffer), pRead) != NULL) {
        printf("%s", buffer);
    }

    fclose(pRead);

    return 0;
}
