typedef struct {
  int id;
  char name[50];
  float value;
} MyStruct;

MyStruct example = {
  #ifdef SPECIAL
  .id = 43,
  #else
  #pragma message("Hello from structure")
  .id = 42,
  #endif
  .name = "Example Name",
  .value = 3.14f,
};

int main() {
  printf("ID: %d\n", example.id);

  return 0;
}
