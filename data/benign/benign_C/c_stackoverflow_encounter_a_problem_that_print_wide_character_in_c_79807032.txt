#include "snake.h"

void SetPos(short x, short y)
{
    //获得标准输出设备的句柄
    HANDLE houtput = NULL;
    houtput = GetStdHandle(STD_OUTPUT_HANDLE);

    //定位光标的位置
    COORD pos = { x, y };
    SetConsoleCursorPosition(houtput, pos);
}
  
void CreateMap()
{
    //上
    int i = 0;
    const int WIDTH = 29;
    const int HEIGHT = 27;

    SetPos(0, 0);
    for (i = 0; i < WIDTH; i++)
    {
        wprintf(L"%lc", L'□');
    }
    //下
    SetPos(0, HEIGHT-1);
    for (i = 0; i < WIDTH; i++)
    {
        wprintf(L"%lc", L'□');
    }
    //左
    for (i = 1; i < HEIGHT-1; i++)
    {
        SetPos(0, i);
        wprintf(L"%lc", L'□');
    }
    //右
    for (i = 1; i < HEIGHT-1; i++)
    {
        SetPos((WIDTH-1)*2, i);
        wprintf(L"%lc", L'□');
    }
    getchar();
}
