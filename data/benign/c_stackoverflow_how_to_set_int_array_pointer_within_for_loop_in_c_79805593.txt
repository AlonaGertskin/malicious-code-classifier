int main()
{
    int *test[3];
        
    /** Try to set the array value one by one */
        
    for (int i = 0; i < 3; i++)
    {
        test[i] = &i;
    }
        
    /** Try to print the array value one by one */
    
    for (int i = 0; i < 3; i++)
    {
        int value = *test[i];
        printf("%d = %d\n", i, value);
    }
        
    return 0;
}
