# Use official Node.js image (stable, small)
FROM node:20-slim

# Prevent interactive prompts in apt and other fixes
ENV DEBIAN_FRONTEND=noninteractive

# Fix potential apt caching/proxy issues
RUN { \
      echo 'Acquire::http::Pipeline-Depth "0";'; \
      echo 'Acquire::http::No-Cache "true";'; \
      echo 'Acquire::BrokenProxy    "true";'; \
    } > /etc/apt/apt.conf.d/99fixbadproxy

# Install system dependencies
RUN apt-get clean && rm -rf /var/lib/apt/lists/* && \
    apt-get update && \
    apt-get install -y --no-install-recommends \
       ca-certificates \
       python3 \
       python3-venv \
       python3-pip \
       build-essential \
       curl && \
    apt-get clean && rm -rf /var/lib/apt/lists/*

# Create non‑root user early
RUN groupadd -r appuser && useradd -r -g appuser -m appuser

WORKDIR /app

# Copy manifest files for Node & Python
COPY package*.json ./
COPY package-lock.json ./
COPY requirements.txt ./

# Install Node dependencies (production only)
ENV NODE_ENV=production
RUN npm install --only=production

# Create Python virtual environment under /opt/venv
RUN python3 -m venv /opt/venv && \
    /opt/venv/bin/pip install --no-cache-dir --upgrade pip && \
    if [ -f requirements.txt ]; then /opt/venv/bin/pip install --no-cache-dir -r requirements.txt; fi

# Make sure /opt/venv/bin is on PATH
ENV PATH="/opt/venv/bin:$PATH"

# Copy full application code
COPY . .

# Create required directories and set permissions
RUN mkdir -p /app/database /app/logs /app/public && \
    chown -R appuser:appuser /app /opt/venv

# Use non‑root user from here onward
USER appuser

# Set runtime environment variables
ENV PORT=8288
EXPOSE 8288

HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD node -e "require('http').get('http://localhost:8288/health', (res) => { process.exit(res.statusCode === 200 ? 0 : 1) })"

# Start the application
CMD ["node", "server.js"]
