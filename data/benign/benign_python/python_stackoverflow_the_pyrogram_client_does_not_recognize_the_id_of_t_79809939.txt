import asyncio
from pyrogram import Client

async def main():
  worker = Client()

  await worker.start()
  chat = await worker.create_channel("New chanel")
  link = await worker.export_chat_invite_link(chat.id)
  print(link)
  print(chat.id)

if __name__=='__main__':
  asyncio.run(main())
